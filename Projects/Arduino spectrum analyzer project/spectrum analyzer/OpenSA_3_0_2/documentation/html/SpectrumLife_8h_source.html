00064 // One step in the Game of Life is to count the number of neighbors that 
00065 // are "on". This function computes part of that: it counts the number of 
00066 // neighbors that are "on" at the current position (r,c) and also the
00067 // row above and below. This is not a Macro because the Game of Life 
00068 // implementation here was ported from an implementation with much less
00069 // flash memory. When you make a macro, that code is expanded with simple
00070 // text substitution before it even gets to the compiler. This means that
00071 // each time we "call" a macro a bunch of code gets included. Somtimes this
00072 // can be optimized out, but not always. By making this a function, we help
00073 // the compiler understand that each "call" can be computed using the same
00074 // code, and this resulted in a smaller binary. TLDR: esoteric design choice
00075 // for size optimization, IGNORE.
00076 #define columnCount(r,c) (get(disp,prevRow(r),c) + get(disp,r,c) + get(disp,nextRow(r),c))

00105 void life() {   
00106         uint8_t i,j,r,c;
00107     // Run the Game of Life
00108     // There are some tedious optimizations here. 
00109     // To compute the next frame in the Game of Life, one has to sum up
00110     // the numbr of "live" or "on" neighbors around each cell.
00111     // To optimize this, we keep a running count. 
00112     // When we want to count the number of "live" cells at position (r,c),
00113     // We look at how many cells were alive around position (r,c-1). Then, we 
00114     // Subtract the cells from column c-2 and add the cells from column c+1.
00115     // One could also keep a running count along the columns, but this
00116     // would require more intermediate state and the additional memory lookups
00117     // consume the benefit in this application. 
00118     // As we update the game, we also keep track of whetehr a cell changes. 
00119     // If cells are not changing, we make note of this, and if the game stays
00120     // stuck for a bit, then we add in new life-forms to keep things
00121     // interesting. 
00122     // One neat thing about double-buffering the game state is that the 
00123     // state from TWO frames ago is available in the output buffer, at least,
00124     // until we write the next frame over it. So we can take advantage of this
00125     // and also detect when the game gets stuck in a 2-cycle.
00126     uint8_t changed = 0;
00127     uint8_t k = 0;
00128     for (r=0; r<NROWS; r++)
00129     {
00130         uint8_t previous = columnCount(r,NCOLS-1);
00131         uint8_t current  = columnCount(r,0);
00132         uint8_t neighbor = previous+current;
00133         for (c=0; c<NCOLS; c++)
00134         {
00135             uint8_t cell = get(disp,r,c);
00136 
00137             uint8_t upcoming = columnCount(r,nextCol(c));
00138             neighbor += upcoming;
00139             uint8_t newcell = cell? (neighbor+1>>1)==2:neighbor==3;
00140 
00141             neighbor-= previous;
00142             previous = current ;
00143             current  = upcoming;
00144             changed |= newcell!=cell && newcell!=get(buff,r,c);
00145             k += newcell;
00146             set(buff,r,c,newcell);
00147         }
00148     }
00149     
00150     uint8_t l=0;
00151     if (!(k&&entropy())) l=genesis;
00152     if (entropy()<5) l=bomb;
00153     if (!changed && shownFor++>TIMEOUT) l=genesis;
00154     if (l) {
00155         uint8_t r = entropy()%NROWS;
00156         uint8_t q = entropy()%NCOLS;
00157         uint8_t a = entropy()&1;
00158         uint8_t b = entropy()&1;

00160         for (i=0;i<3;i++)
00161         {    
00162             uint8_t c = q;
00163             for (j=0;j<3;j++) 
00164             {
00165                 set(buff,r,c,(l&0x80)>>7?1:0);
00166                 l <<= 1;
00167                 c = a?nextCol(c):prevCol(c);
00168             }
00169             r = b?nextRow(r):prevRow(r);
00170         }

00172     }
